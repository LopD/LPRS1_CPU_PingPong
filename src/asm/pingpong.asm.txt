count_frames_begin:
	sub R0,R0,R0
	inc R0,R0
	ld R1, R0 ;; R1 <- frame_cnt
	inc R0,R0 ;;
	ld R2, R0 ;; R2 <- frames_per_heartbeat
	dec R0,R0
	inc R1,R1 ;; frame_cnt++

	jmpz count_frames_heartbeat
	st R1, R0 ;; R1 -> frame_cnt
	jmp frame_sync_rising_edge
